### Download compiler from https://github.com/protocolbuffers/protobuf/releases

cd tensorflow

set SRC_DIR=D:/Projects/tensorflow
set DST_DIR=D:/SciSharp/TensorFlow.NET/src/TensorFlowNET.Core/Protobuf
